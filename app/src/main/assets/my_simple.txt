# method wait loading to home screen
fun waitLoadingToHomeScreen
    delay 2000
    findImagePosition /sdcard/Download/loading.png iconX iconY confidence
    if $confidence > 0.8
        logs High confidence detection
    else
        logs Low confidence, skipping
        call waitLoadingToHomeScreen
    endif
endfun

    endif
endfun
